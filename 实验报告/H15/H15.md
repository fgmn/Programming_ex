#include<iostream>
#include<vector>
using namespace std;
using ll = long long;

const int Mod = 1e9 + 7;
int C[15][15];

struct mat
{
	int n;
	vector<vector<long long>> x;

	// æ„é€ å‡½æ•°
	explicit mat(int sz) : x(sz, vector<long long>(sz)) { n = sz; }

	// å•ä½çŸ©é˜µ
	void emat()
	{
		for (int i = 0; i < n; i++) {
			x[i][i] = 1;
		}
	}

	// é‡è½½ä¹˜å·
	mat operator*(const mat &a) const
	{
		mat res(n);
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < n; j++) {
				for (int k = 0; k < n; k++) {
					res.x[i][j] = (res.x[i][j] + x[i][k] * a.x[k][j]) % Mod;
				}
			}
		}
		return res;
	}
};

mat matqp(mat a, long long b)

    int n, k;
    cin >> n >> k;

    // ç»„åˆæ•°è®¡ç®— C00 = 0

            C[i][j] = C[i - 1][j - 1] + C[i - 1][j];
